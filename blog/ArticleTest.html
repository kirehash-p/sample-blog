const groupBy = <T, K extends string | number>(items: T[], key: (item: T) => K) => {
  return items.reduce<Record<K, T[]>>((acc, item) => {
    const k = key(item);
    if (!acc[k]) acc[k] = [];
    acc[k].push(item);
    return acc;
  }, {} as Record<K, T[]>);
};
